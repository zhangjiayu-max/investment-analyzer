def _get_technical_score(theme_rule: dict) -> tuple[int, str]:
    """获取主题对应指数的技术指标得分。
    
    Returns:
        (score_delta, signal): score_delta 范围 -10~+15，signal 为 "bull"/"bear"/"neutral"
    """
    try:
        # 找到主题对应的指数代码
        index_code = _get_theme_index_code(theme_rule)
        if not index_code:
            return 0, "neutral"
        
        import akshare as ak
        # 获取近 60 日收盘价
        df = ak.stock_zh_index_daily(symbol=index_code)
        if df is None or len(df) < 30:
            return 0, "neutral"
        
        closes = df['close'].values[-60:]
        
        # 1. MACD 简化版
        ema12 = _ema(closes, 12)
        ema26 = _ema(closes, 26)
        macd = ema12 - ema26
        signal_line = _ema(macd, 9) if len(macd) >= 9 else None
        macd_bull = signal_line is not None and macd[-1] > signal_line[-1]
        
        # 2. RSI(14)
        rsi = _rsi(closes, 14)
        rsi_bull = 30 < rsi < 70  # 中性偏强
        
        # 3. 均线多头排列
        ma5 = closes[-5:].mean()
        ma20 = closes[-20:].mean()
        ma_bull = ma5 > ma20 and closes[-1] > ma20
        
        # 综合评分
        score = 0
        if macd_bull:
            score += 5
        if rsi_bull:
            score += 5
        if ma_bull:
            score += 5
        
        signal = "bull" if score >= 10 else ("bear" if score == 0 else "neutral")
        return score, signal
    except Exception as e:
        logger.debug(f"[opportunity] 技术指标获取失败: {e}")
        return 0, "neutral"
```

def review_opportunity_tracks() -> dict:
    """批量回测已到期的机会跟踪记录（超过 review_date 但 hit 为 None）。"""
    from db.opportunities import list_pending_review_tracks, update_opportunity_track
    pending = list_pending_review_tracks()
    reviewed = 0
    hit_count = 0
    for track in pending:
        try:
            review_price = _get_theme_current_price_by_date(track["theme"], track["review_date"])
            if review_price is None:
                continue
            entry_price = track["entry_price"]
            change_pct = (review_price - entry_price) / entry_price * 100
            # 命中定义：15 交易日后涨幅 >= 3%（考虑交易成本）
            hit = change_pct >= 3.0
            update_opportunity_track(track["id"], {
                "review_price": review_price,
                "hit": hit,
                "change_pct": round(change_pct, 2),
            })
            reviewed += 1
            if hit:
                hit_count += 1
        except Exception as e:
            logger.warning(f"[opportunity] 回测失败 {track['id']}: {e}")
    
    return {"reviewed": reviewed, "hit": hit_count, "miss": reviewed - hit_count}

    change_pct REAL,
    created_at TEXT,
    FOREIGN KEY (opportunity_id) REFERENCES theme_opportunities(id)
);
```

## 七、风险评估

### 潜在风险

1. **P1-K/L/M 接入新维度可能拖慢响应**
   - 每个维度调用 akshare 一次，可能增加 1-3s
   - 缓解：用 `_call_akshare_with_timeout` 包裹，超时返回 0 分

2. **P0-I 去重放宽可能误判不同事件**
   - 同主题不同事件（如"美联储加息"vs"美联储降息"）可能被误判
   - 缓解：标题相似度阈值保持 0.6，仅日期放宽到 ±3 天

3. **P1-N 回测依赖 akshare 历史数据**
   - 可能因节假日/停牌获取不到 review_date 当日价格
   - 缓解：用最近交易日价格 fallback

4. **P0-A 估值过高一票否决可能误伤周期股**
   - 周期股 PE 高时可能是底部（钢铁/有色）
   - 缓解：在 verdict 降级时记录 `valuation_warning` 字段提示"估值过高"，让用户/专家自行判断

### 不影响现有功能的保障

- 所有新维度都包 try/except，失败返回 0 分不影响主流程
- P0-F 仅更新配置表，不影响代码逻辑
- P0-G/H/I/J 都在事件雷达路径，独立于对话主流程
