### 1.2 断裂点（需要胶水）

```
断裂 1: 对话结束后不自动触发评测
  → process_conversation_evaluation() 定义了但从未被调用
  → 位置: orchestrator.py / conversations.py 末尾

断裂 2: 每日评测没有 cron
  → run_daily_eval() 只通过手动 API 触发
  → 位置: app.py startup

断裂 3: Shadow 胜率被统计但不自动晋升
  → shadow_win_rate 只展示，不自动 activate
  → 位置: shadow_mode.py

断裂 4: 根因分析给建议但不生成 prompt 变体
  → suggestion 字段只存不读
  → 位置: root_cause_analyzer.py → eval_system.py

断裂 5: 自动 Eval 生成不自动调度
  → auto_generate_eval_cases() 只通过手动调用
  → 位置: auto_eval_generator.py
```

---

## 2. 设计目标

1. **对话即评测** — 每次 AI 对话结束后自动评估质量
2. **低分即修复** — 连续低分触发根因分析 → 自动生成 prompt 变体 → 进入 Shadow
3. **高分即基准** — 高分对话自动转为 Eval 用例
4. **Shadow 自动晋升** — 连续 N 天胜率 >60% 自动替换 active prompt
5. **每日自动运转** — 每日评测 + Eval 生成 + Shadow 晋升在 cron 中自动执行

### 3.2 评测调度器

```python
# orchestrator.py 新增
import asyncio
from concurrent.futures import ThreadPoolExecutor

_eval_executor = ThreadPoolExecutor(max_workers=1, thread_name_prefix="eval")


def _schedule_auto_evaluation(conv_id: int, msg_id: int, result: dict):
    """异步调度对话质量评测（不阻塞主流程）。"""
    _eval_executor.submit(
        _run_auto_evaluation_sync, conv_id, msg_id, result
    )


def _run_auto_evaluation_sync(conv_id: int, msg_id: int, result: dict):
    """在独立线程中运行评测，避免阻塞 orchestrator 线程池。"""
    try:
        loop = asyncio.new_event_loop()
        loop.run_until_complete(
            _do_auto_evaluation(conv_id, msg_id, result)
        )
    except Exception as e:
        logger.error(f"自动评测失败: conv={conv_id} msg={msg_id} — {e}")
    finally:
        loop.close()


async def _do_auto_evaluation(conv_id: int, msg_id: int, result: dict):
    """执行自动评测：规则评估 → 低分走进化流程。"""
    from agent.conversation_evaluator import ConversationQualityEvaluator
    from agent.conversation_evolution import process_conversation_evaluation
    
    evaluator = ConversationQualityEvaluator()
    evaluation = evaluator.evaluate(
        conv_id, msg_id, 
        trigger_evolution=False,  # 我们手动控制触发
        use_llm=False             # 先规则评估，快
    )
    
    # 转换成 process_conversation_evaluation 需要的格式
    ev = {
        "auto_score": evaluation.auto_score,
        "auto_score_breakdown": {
            dim.name: {"score": dim.score, "details": dim.details}
            for dim in evaluation.dimensions
        },
        "suggestions": evaluation.suggestions,
    }
    
    # 🎯 核心：触发进化流程
    await process_conversation_evaluation(conv_id, ev)
    
    # 如果规则评估分低，跑一次 LLM 评估做更深分析
    if evaluation.auto_score < LOW_SCORE_THRESHOLD:
        llm_eval = evaluator.evaluate(
            conv_id, msg_id, 
            trigger_evolution=False,
            use_llm=True
        )
        await process_conversation_evaluation(conv_id, {
            "auto_score": llm_eval.auto_score,
            "auto_score_breakdown": llm_eval.auto_score_breakdown,
            "suggestions": llm_eval.suggestions,
        })
```

## 4. 胶水 2：每日评测 Cron

### 4.1 定时任务

```python
# app.py startup() 中新增
asyncio.create_task(_auto_daily_eval())

async def _auto_daily_eval():
    """每日凌晨 4:00 自动执行评测 Pipeline。"""
    await asyncio.sleep(10)  # 等启动完成
    
    while True:
        now = datetime.now()
        # 计算到凌晨 4:00 的秒数
        target = now.replace(hour=4, minute=0, second=0, microsecond=0)
        if now >= target:
            target += timedelta(days=1)
        wait_seconds = (target - now).total_seconds()
        
        await asyncio.sleep(wait_seconds)
        
        try:
            logger.info("=== 每日自动评测开始 ===")
            
            # Step 1: 跑每日评测
            from routers.analysis.eval_system import run_daily_eval
            eval_result = await run_daily_eval()
            logger.info(f"每日评测完成: avg={eval_result.get('avg_score')}")
            
            # Step 2: 检查并晋升 Shadow
            from shadow_mode import auto_promote_shadows as _auto_promote
            promoted = await _auto_promote()
            logger.info(f"Shadow 晋升: {promoted}")
            
            # Step 3: 自动生成 Eval 用例
            from auto_eval_generator import auto_generate_eval_cases
            gen_result = auto_generate_eval_cases()
            logger.info(f"Eval 用例生成: {gen_result}")
            
            logger.info("=== 每日自动评测完成 ===")
        except Exception as e:
            logger.error(f"每日评测失败: {e}")
```

## 5. 胶水 3：自动根因+修复+Shadow

### 5.1 增强 conversation_evolution.py

在 `process_conversation_evaluation` 中，低分对话增加自动修复链：

```python
# conversation_evolution.py 中 _trigger_feedback_learning 之后
async def _auto_root_cause_to_fix(conversation_id, evaluation):
    """低分 → 根因 → 自动生成 prompt 修复 → 进入 Shadow。"""
    from root_cause_analyzer import analyze_root_cause, batch_analyze
    from db.eval import create_prompt_version, get_active_prompt
    from shadow_mode import create_shadow_config
    
    # 1. 收集最近的 Bad Case
    bad_cases = list_all_bad_cases(limit=20)
    if not bad_cases:
        return None
    
    # 2. 批量根因分析
    analysis = batch_analyze(limit=20, force=False)
    
    # 3. 找出最频繁的根因（至少出现 3 次）
    freq = {}
    for case_id, result in analysis.get("results", {}).items():
        rc = result.get("root_cause", "")
        if rc and rc != "other":
            freq.setdefault(rc, []).append(result)
    
    top_causes = sorted(freq.items(), key=lambda x: -len(x[1]))
    fixes_applied = []
    
    for root_cause, results in top_causes:
        if len(results) < 3:
            continue  # 至少 3 个 case 指向同一根因才自动修复
        
        # 4. 找到受影响最严重的 agent_type
        agent_type = _infer_agent_type_from_bad_cases(results)
        if not agent_type:
            continue
        
        # 5. 获取当前 active prompt
        active = get_active_prompt(agent_type)
        if not active:
            continue
        
        # 6. 合并多个 case 的建议，生成改进 prompt
        suggestions = list(set(
            r.get("suggestion", "") for r in results if r.get("suggestion")
        ))
        improved_prompt = _generate_improved_prompt(
            active["prompt_content"], 
            root_cause, 
            suggestions,
            results
        )
        
        # 7. 创建新 prompt 版本（draft，不激活）
        version_id = create_prompt_version(
            agent_type=agent_type,
            version=f"auto-fix-{datetime.now():%Y%m%d%H%M}",
            prompt_content=improved_prompt,
            changelog=f"自动修复: {root_cause} ({len(results)}个Bad Case) — "
                      f"{'; '.join(s[:50] for s in suggestions[:3])}",
            is_active=0,
        )
        
        # 8. 创建 Shadow config（自动对比新旧版本）
        shadow_id = create_shadow_config(
            name=f"auto-fix: {agent_type} {root_cause}",
            agent_type=agent_type,
            current_prompt=active["prompt_content"],
            candidate_prompt=improved_prompt,
            traffic_pct=0.1,
        )
        
        fixes_applied.append({
            "root_cause": root_cause,
            "case_count": len(results),
            "version_id": version_id,
            "shadow_id": shadow_id,
        })
        
        logger.info(
            f"自动修复: {agent_type} {root_cause} → "
            f"v{version_id} → shadow#{shadow_id}"
        )
    
    return fixes_applied
```

### 5.2 Prompt 自动改进函数

```python
def _generate_improved_prompt(
    current_prompt: str, 
    root_cause: str, 
    suggestions: list[str],
    bad_cases: list[dict]
) -> str:
    """让 LLM 基于根因和建议，生成改进版的 prompt。"""
    from llm_service import _call_llm
    
    cases_text = "\n\n".join(
        f"Bad Case #{i+1}: {c.get('detail', '')}\n证据: {c.get('evidence', '')}"
        for i, c in enumerate(bad_cases[:5])
    )
    
    prompt = f"""你是一个 Prompt 工程专家。以下是当前 Agent prompt 和它的 Bad Cases。
请基于根因分析结果，改进 prompt 以避免同类问题。

## 当前 Prompt
{current_prompt[:3000]}

## 根因
{root_cause}

## 改进建议
{chr(10).join(f'- {s}' for s in suggestions[:5])}

## Bad Cases
{cases_text[:2000]}

## 要求
1. 保持原有 prompt 的结构和角色定义
2. 在关键位置增加约束或示例，针对性解决上述问题
3. 不要删除原有内容，只增加和修改
4. 直接输出改进后的完整 prompt
"""

    response = _call_llm(
        caller="prompt_improver",
        model=MODEL,
        messages=[{"role": "user", "content": prompt}],
        temperature=0.3,
        max_tokens=8000,
    )
    
    return response.choices[0].message.content or current_prompt
```

### 5.3 推断影响最大的 agent_type

```python
def _infer_agent_type_from_bad_cases(results: list[dict]) -> str | None:
    """从 Bad Case 中推断哪个 agent 受此根因影响最大。"""
    from db.eval import list_eval_cases
    
    # 统计各 agent_type 的 Bad Case 数量
    agent_counts = {}
    for r in results:
        # Bad Case 可能关联了 eval case
        # 从 eval case 的 analysis_type 推断
        agent_type = r.get("agent_type", "")
        if not agent_type:
            # 尝试从 detail 推断
            detail = r.get("detail", "")
            if "估值" in detail:
                agent_type = "valuation_expert"
            elif "风险" in detail:
                agent_type = "risk_assessor"
            elif "配置" in detail:
                agent_type = "allocation_advisor"
            elif "市场" in detail:
                agent_type = "market_analyst"
            else:
                agent_type = "general"
        
        agent_counts[agent_type] = agent_counts.get(agent_type, 0) + 1
    
    if not agent_counts:
        return None
    
    return max(agent_counts, key=agent_counts.get)
```

## 6. 胶水 4：Shadow 自动晋升

### 6.1 shadow_mode.py 新增

```python
# shadow_mode.py 新增
async def auto_promote_shadows() -> list[dict]:
    """自动检查所有活跃 shadow config，符合条件的自动晋升。

    晋升条件：
    - 至少 30 次对比（足够样本量）
    - shadow 胜率 > 60%
    - shadow 平均分 > production 平均分 + 0.5
    - 最近 7 天有对比数据（仍在活跃使用）
    """
    from db.eval import activate_prompt_version
    
    configs = list_shadow_configs(active_only=True)
    promoted = []
    
    for cfg in configs:
        stats = get_shadow_stats(cfg["id"])
        
        total_runs = stats.get("total_runs", 0)
        if total_runs < 30:
            continue
        
        win_rate = stats.get("shadow_win_rate", 0) or 0
        avg_prod = stats.get("avg_prod_score", 0) or 0
        avg_shadow = stats.get("avg_shadow_score", 0) or 0
        
        # 检查晋升条件
        if win_rate > 0.6 and (avg_shadow - avg_prod) > 0.5:
            # 需要知道这个 shadow config 关联了哪个 prompt version
            # 从 description 中解析，或者在创建时记录 version_id
            
            # 激活对应的 prompt 版本
            activate_prompt_version(
                cfg.get("prompt_version_id"),
                cfg["agent_type"]
            )
            
            # 停用 shadow config
            toggle_shadow_config(cfg["id"], is_active=False)
            
            logger.info(
                f"Shadow 自动晋升: {cfg['agent_type']} #{cfg['id']} "
                f"({total_runs} runs, win_rate={win_rate:.1%}, "
                f"shadow={avg_shadow:.1f} > prod={avg_prod:.1f})"
            )
            
            promoted.append({
                "config_id": cfg["id"],
                "agent_type": cfg["agent_type"],
                "runs": total_runs,
                "win_rate": round(win_rate, 3),
                "shadow_score": round(avg_shadow, 2),
                "prod_score": round(avg_prod, 2),
                "action": "promoted",
            })
    
    return promoted
```

## 10. 验收标准

| 场景 | 预期行为 |
|------|---------|
| 用户完成一次对话 | 后台自动运行规则评估，低分时追加 LLM 评估 |
| 3 个以上 Bad Case 指向同一根因 | 自动生成改进 prompt → 创建 Shadow config |
| Shadow config 有 30+ 次对比，胜率 >60% | 每日 cron 自动晋升为 active prompt |
| 每日 04:00 | 自动跑评测 + Shadow 晋升检查 + Eval 用例生成 |
| 高分对话 (≥85分) | 自动转为 Eval 基准用例 |
| 同一消息被重复评测 | `has_evaluation` 返回 True，跳过 |
