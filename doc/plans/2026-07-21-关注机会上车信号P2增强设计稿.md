#### 3.2.1 自适应阈值调整

新建 `backend/services/advisor/watchlist_adaptive.py`：

```python
"""关注列表自适应阈值调整。

根据历史命中率动态调整 target_percentile：
- hit_rate >= 70%（高命中率）：放宽 10%（更多机会触发）
- hit_rate < 40%（低命中率）：收紧 20%（更难触发，避免误信号）
- 40% <= hit_rate < 70%：不调整
- reviewed < 5（样本不足）：不调整

 Returns:
    (adjusted_target_pct, adjustment_reason)
"""
from db.config import get_config_bool, get_config_int
from db.watchlist import get_signal_backtest_stats


def apply_adaptive_threshold(fund_code: str, target_pct: float | None) -> tuple[float | None, str]:
    """根据历史命中率自适应调整 target_percentile 阈值。"""
    try:
        if not get_config_bool("watchlist.adaptive_threshold_enabled", True):
            return target_pct, ""
    except Exception:
        pass

    if not fund_code or target_pct is None:
        return target_pct, ""

    try:
        min_samples = get_config_int("watchlist.adaptive_min_samples", 5)
    except Exception:
        min_samples = 5

    stats = get_signal_backtest_stats(fund_code=fund_code)
    reviewed = stats.get("reviewed", 0)
    if reviewed < min_samples:
        return target_pct, ""

    hit_rate = stats.get("hit_rate") or 0
    if hit_rate >= 70:
        adjusted = round(target_pct * 1.1, 2)
        return adjusted, f"命中率 {hit_rate:.0f}% 偏高，阈值放宽至 {adjusted}%"
    elif hit_rate < 40:
        adjusted = round(target_pct * 0.8, 2)
        return adjusted, f"命中率 {hit_rate:.0f}% 偏低，阈值收紧至 {adjusted}%"
    return target_pct, ""
```

#### 4.2.1 新建宏观信号模块

新建 `backend/services/advisor/watchlist_macro.py`：

```python
"""关注列表宏观信号维度。

5 类宏观指标（akshare 数据源，全部已封装在 tools/__init__.py）：
- LPR（贷款市场报价利率）：1Y < 3.5% 宽松 +5，> 4.0% 收紧 -5
- SHIBOR 3M：< 1.5% 流动性充裕 +5，> 2.5% 偏紧 -5
- 美债 10Y：> 4.0% 全球资金压力 -5，< 3.5% 中性偏松 +3
- USDCNH 汇率：贬值 > 7.3 警惕 -5，升值 < 7.0 利好 +3
- 政策信号：央行降准/降息 +8，加息/提准 -8（来自 services/market/policy_news.py）

输出：
- macro_score: -15 ~ +15
- macro_signal: easing / tightening / neutral
- macro_reasons: List[str]
- macro_details: {lpr, shibor, us_treasury, usdcnh, policy}
"""
import time
import threading
from db.config import get_config_bool

_MACRO_CACHE = {}
_MACRO_CACHE_LOCK = threading.Lock()
_MACRO_CACHE_TTL = 60 * 60  # 1 小时（宏观数据日级更新）


def get_macro_score_cached() -> dict:
    """宏观信号（全局共享，1 小时缓存）。"""
    with _MACRO_CACHE_LOCK:
        cached = _MACRO_CACHE.get("macro")
        if cached and time.time() - cached[0] < _MACRO_CACHE_TTL:
            return cached[1]

    try:
        if not get_config_bool("watchlist.macro_signal_enabled", True):
            return {"score": 0, "signal": "neutral", "reasons": [], "details": {}, "disabled": True}
    except Exception:
        pass

    lpr_score, lpr_reason = _get_lpr_score()
    shibor_score, shibor_reason = _get_shibor_score()
    us_treasury_score, us_treasury_reason = _get_us_treasury_score()
    usdcnh_score, usdcnh_reason = _get_usdcnh_score()
    policy_score, policy_reason = _get_policy_score()

    total_score = (lpr_score + shibor_score + us_treasury_score
                   + usdcnh_score + policy_score)
    if total_score >= 8:
        signal = "easing"
    elif total_score <= -8:
        signal = "tightening"
    else:
        signal = "neutral"

    result = {
        "score": total_score,
        "signal": signal,
        "reasons": [r for r in [lpr_reason, shibor_reason, us_treasury_reason,
                                 usdcnh_reason, policy_reason] if r],
        "details": {
            "lpr": lpr_score, "shibor": shibor_score,
            "us_treasury": us_treasury_score, "usdcnh": usdcnh_score,
            "policy": policy_score,
        },
    }
    with _MACRO_CACHE_LOCK:
        _MACRO_CACHE["macro"] = (time.time(), result)
    return result
```

### 5.2 改造方案

#### 5.2.1 新建共振检测模块

新建 `backend/services/advisor/watchlist_resonance.py`：

```python
"""关注列表 + 持仓 组合层信号共振检测。

两类共振：
1. watchlist 多标的同向共振：N 只关注基金同时 green = 加仓窗口
2. holding 系统性风险预警：N 只持仓同时触发退出信号 = 系统性风险

阈值：
- strong: green_count >= 3 或 green_ratio >= 30%
- moderate: green_count >= 2
- weak/none: 其他
"""
from db.watchlist import list_watchlist
from db.config import get_config_bool, get_config_int, get_config_float


def detect_watchlist_resonance(user_id: str = "default") -> dict:
    """检测关注列表信号共振。"""
    try:
        if not get_config_bool("watchlist.resonance_detection_enabled", True):
            return {"disabled": True}
    except Exception:
        pass

    items = list_watchlist(user_id=user_id, status="watching")
    green = [it for it in items if it.get("signal_status") == "green"]
    yellow = [it for it in items if it.get("signal_status") == "yellow"]
    red = [it for it in items if it.get("signal_status") == "red"]
    total = len(items)

    if total == 0:
        return {
            "green_count": 0, "yellow_count": 0, "red_count": 0, "total": 0,
            "resonance_level": "none", "resonance_type": "neutral",
            "alert_funds": [], "suggestion": "关注列表为空",
        }

    try:
        strong_count = get_config_int("watchlist.resonance_strong_threshold", 3)
    except Exception:
        strong_count = 3
    try:
        strong_ratio = get_config_float("watchlist.resonance_ratio_threshold", 0.3)
    except Exception:
        strong_ratio = 0.3

    green_ratio = len(green) / total
    alert_funds = [{"fund_code": it["fund_code"], "fund_name": it.get("fund_name", ""),
                    "signal_reason": it.get("signal_reason", "")} for it in green]

    if len(green) >= strong_count or green_ratio >= strong_ratio:
        level, suggestion = "strong", f"{len(green)} 只关注基金同时 green，加仓窗口"
    elif len(green) >= 2:
        level, suggestion = "moderate", f"{len(green)} 只关注基金 green，可关注"
    elif len(red) >= strong_count:
        level, suggestion = "strong_bearish", f"{len(red)} 只关注基金 red，市场可能承压"
    else:
        level, suggestion = "weak", "信号分散，无明显共振"

    return {
        "green_count": len(green), "yellow_count": len(yellow),
        "red_count": len(red), "total": total,
        "green_ratio": round(green_ratio, 3),
        "resonance_level": level,
        "resonance_type": "bullish" if len(green) >= 2 else ("bearish" if len(red) >= 2 else "neutral"),
        "alert_funds": alert_funds,
        "suggestion": suggestion,
    }


def detect_holding_systemic_risk(user_id: str = "default") -> dict:
    """检测持仓系统性风险（多只持仓同时触发退出信号）。"""
    try:
        from db.portfolio import list_holdings
        holdings = list_holdings(user_id=user_id)
    except Exception:
        holdings = []

    if not holdings:
        return {"systemic_risk": False, "triggered_count": 0, "total_holdings": 0, "triggered_funds": []}

    exit_triggered = [h for h in holdings if h.get("exit_signal")]
    total = len(holdings)
    triggered_ratio = len(exit_triggered) / total if total else 0

    try:
        strong_count = get_config_int("watchlist.resonance_strong_threshold", 3)
    except Exception:
        strong_count = 3
    try:
        strong_ratio = get_config_float("watchlist.resonance_ratio_threshold", 0.3)
    except Exception:
        strong_ratio = 0.3

    triggered_funds = [{
        "fund_code": h.get("fund_code"), "fund_name": h.get("fund_name"),
        "exit_signal": h.get("exit_signal"), "exit_signal_reason": h.get("exit_signal_reason"),
    } for h in exit_triggered]

    is_systemic = len(exit_triggered) >= strong_count or triggered_ratio >= strong_ratio

    return {
        "systemic_risk": is_systemic,
        "triggered_count": len(exit_triggered),
        "total_holdings": total,
        "triggered_ratio": round(triggered_ratio, 3),
        "triggered_funds": triggered_funds,
        "suggestion": "多只持仓同时触发退出，疑似系统性风险，建议降低仓位" if is_systemic else "",
    }
```

```python
        from db.portfolio import create_alert
        create_alert(
            alert_type="watchlist_signal_change",
            title=change_title,
            content=signal_reason,
            severity=severity_map.get(signal_status, "info"),
            related_fund_code=fund_code,
            related_fund_name=fund_name,
            source="watchlist_patrol",
        )
    except Exception as _e:
        logger.debug(f"[watchlist] 信号变更 alert 写入失败 {fund_code}: {_e}")

    # SSE 推送
    if change_sse_enabled:
        try:
            import asyncio
            from routers.conversation.notifications import notify_subscribers
            asyncio.create_task(notify_subscribers({
                "type": "watchlist_signal_change",
                "title": f"{fund_name} 信号灯变更",
                "message": f"{last_status} → {signal_status}: {signal_reason}",
                "data": {
                    "fund_code": fund_code, "fund_name": fund_name,
                    "from": last_status, "to": signal_status,
                    "reason": signal_reason,
                },
            }))
        except Exception as _e:
            logger.debug(f"[watchlist] SSE 推送失败 {fund_code}: {_e}")
```

```vue
<script setup>
import { ref, onMounted, onUnmounted } from 'vue'

const notifications = ref([])
const showPanel = ref(false)
let eventSource = null

onMounted(() => {
  const protocol = window.location.protocol === 'https:' ? 'wss' : 'ws'
  const host = window.location.host
  eventSource = new EventSource(`${protocol}//${host}/api/notifications/stream`.replace('wss', 'https').replace('ws', 'http'))
  eventSource.onmessage = (event) => {
    try {
      const data = JSON.parse(event.data)
      notifications.value.unshift(data)
      if (notifications.value.length > 50) notifications.value.pop()
    } catch (e) {}
  }
  eventSource.onerror = () => {
    setTimeout(() => eventSource && eventSource.close() && initEventSource(), 5000)
  }
})

onUnmounted(() => eventSource && eventSource.close())
</script>
```

注：protocol 处理用 `window.location.protocol === 'https:' ? 'https' : 'http'`（SSE 走 HTTP/HTTPS，不是 ws/wss）。

## 十、风险与降级

1. **宏观信号数据源失败**：每个指标独立 try/except，失败时该指标 0 分，不影响其他指标；1 小时缓存避免频繁调用
2. **自适应阈值样本不足**：`reviewed < 5` 时不调整，维持原 target_pct；新基金冷启动期无反哺
3. **共振检测误判**：仅在 `green >= 3` 或 `ratio >= 30%` 时触发 strong，避免小样本误判；空列表返回 none
4. **SSE 推送失败**：try/except 兜底，不影响 patrol 主流程；前端 EventSource 自动重连
5. **信号变更 alert 去重**：title 含 `from → to` 标识，天然区分不同变更方向；同方向 24 小时内去重避免刷屏
6. **前端轨迹图无数据**：展示空状态提示"暂无回测数据"
7. **配置项注册冲突**：使用 `INSERT OR IGNORE`，重启自动注册，不覆盖已有值
8. **宏观权重调整影响历史可比性**：confidence 计算逻辑变化，但历史 signal_confidence 已写入回测表，不影响已记录数据
