<script setup>
import { computed } from 'vue'
import Icon from './Icon.vue'

const props = defineProps({
  label: { type: String, required: true },
  agent: { type: String, default: '' },
  icon: { type: String, default: 'bot' },
  loading: { type: Boolean, default: false },
  disabled: { type: Boolean, default: false },
  variant: { type: String, default: 'primary' },
  size: { type: String, default: 'md' },
  state: { type: String, default: 'idle' },
})

const emit = defineEmits(['click'])

const STATE_CONFIG = {
  idle: { label: '', icon: '', disabled: false, loading: false },
  submitting: { label: '提交中...', icon: 'hourglass', disabled: true, loading: true },
  queued: { label: '已提交后台分析', icon: 'check', disabled: true, loading: false },
  running: { label: '分析中...', icon: 'hourglass', disabled: true, loading: true },
  error: { label: '重试分析', icon: 'refresh', disabled: false, loading: false },
}

const effectiveState = computed(() => {
  if (props.loading) return 'running'
  return STATE_CONFIG[props.state] ? props.state : 'idle'
})

const stateConfig = computed(() => STATE_CONFIG[effectiveState.value])
const isBusy = computed(() => stateConfig.value.loading)
const isDisabled = computed(() => props.disabled || stateConfig.value.disabled)
const displayLabel = computed(() => stateConfig.value.label || props.label)
const displayIcon = computed(() => stateConfig.value.icon || props.icon)
</script>

<template>
  <button
    type="button"
    :class="[
      'ai-action-button',
      `ai-action-button--${variant}`,
      `ai-action-button--${size}`,
      `ai-action-button--${effectiveState}`,

  description: { type: String, default: '' },

            @click="emit('action')"
          >
            {{ actionText }}
          </button>
        </div>
      </div>
    </div>
    <slot v-else />
  </div>
</template>
