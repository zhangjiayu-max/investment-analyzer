# 对话评估与进化系统集成方案

> 日期：2026-06-12 | 状态：✅ 已完成

## 一、目标

将对话质量评估系统与现有进化系统深度集成，形成**自动化的质量改进闭环**：

```
对话完成 → 自动评估 → 低分信号 → 反馈学习 + Bad Case
                ↓
        高分对话 → 转化为 Eval 基准
                ↓
        评估洞察 → Prompt 优化 + 专家调优
                ↓
        下次对话质量提升 → 持续迭代
```

## 三、集成设计

### 3.1 低分对话自动触发反馈学习

**思路**：当对话评估分数低于阈值时，自动调用 feedback_learner 更新用户画像。

**实现**：

```python
# backend/agent/conversation_evolution.py

LOW_SCORE_THRESHOLD = 60  # 低于60分触发
HIGH_SCORE_THRESHOLD = 85  # 高于85分触发

async def process_conversation_evaluation(conversation_id: int, evaluation: dict):
    """处理对话评估结果，触发进化机制"""
    auto_score = evaluation.get("auto_score", 0)
    breakdown = evaluation.get("auto_score_breakdown", {})
    suggestions = evaluation.get("suggestions", [])

    # 1. 低分对话 → 反馈学习
    if auto_score < LOW_SCORE_THRESHOLD:
        await _trigger_feedback_learning(conversation_id, evaluation)

    # 2. 低分对话 → 自动标记 Bad Case
    if auto_score < LOW_SCORE_THRESHOLD:
        await _auto_mark_bad_case(conversation_id, evaluation)

    # 3. 高分对话 → 转化为 Eval 基准
    if auto_score >= HIGH_SCORE_THRESHOLD:
        await _suggest_as_eval_case(conversation_id, evaluation)

    # 4. 评估洞察 → 专家调优建议
    await _analyze_expert_performance(conversation_id, evaluation)


async def _trigger_feedback_learning(conversation_id: int, evaluation: dict):
    """低分对话触发反馈学习"""
    from agent.feedback_learner import update_user_profile_from_feedback
    from db.conversations import get_messages

    messages = get_messages(conversation_id)
    if not messages:
        return

    # 提取用户问题和助手回答
    user_query = ""
    assistant_response = ""
    for msg in messages:
        if msg["role"] == "user" and not user_query:
            user_query = msg["content"][:200]
        elif msg["role"] == "assistant":
            assistant_response = msg["content"][:500]

    # 构建反馈信号
    breakdown = evaluation.get("auto_score_breakdown", {})
    weak_dimensions = [k for k, v in breakdown.items() if v < 60]

    note = f"对话质量评分: {evaluation['auto_score']:.0f}/100。"
    if weak_dimensions:
        dim_names = {
            "execution": "执行效率",
            "data": "数据利用",
            "collaboration": "专家协作",
            "response": "响应质量"
        }
        note += f"薄弱维度: {', '.join(dim_names.get(d, d) for d in weak_dimensions)}。"

    suggestions = evaluation.get("suggestions", [])
    if suggestions:
        note += f"建议: {suggestions[0]}"

    # 触发反馈学习
    update_user_profile_from_feedback(
        user_id="default",
        feedback_type="unhelpful",
        note=note,
        input_summary=user_query,
    )

    logger.info(f"对话 {conversation_id} 低分触发反馈学习: {evaluation['auto_score']:.0f}分")
```

### 3.2 低分对话自动标记 Bad Case

**思路**：低分对话自动记录到 llm_feedback 表，供 Bad Case 看板展示。

**实现**：

```python
async def _auto_mark_bad_case(conversation_id: int, evaluation: dict):
    """低分对话自动标记为 Bad Case"""
    from db import save_llm_feedback

    breakdown = evaluation.get("auto_score_breakdown", {})

    # 计算三维度分数（映射到 1-10 分制）
    data_score = min(10, max(1, breakdown.get("data", 50) / 10))
    logic_score = min(10, max(1, breakdown.get("collaboration", 50) / 10))
    action_score = min(10, max(1, breakdown.get("response", 50) / 10))
    overall_score = min(10, max(1, evaluation.get("auto_score", 50) / 10))

    # 构建输入摘要
    from db.conversations import get_messages
    messages = get_messages(conversation_id)
    input_summary = ""
    for msg in messages:
        if msg["role"] == "user":
            input_summary = msg["content"][:200]
            break

    # 保存到 llm_feedback 表
    save_llm_feedback(
        caller="conversation_eval",
        input_summary=input_summary,
        output_summary=f"对话质量评分: {evaluation['auto_score']:.0f}/100",
        rating="unhelpful",
        tags=",".join(evaluation.get("suggestions", [])[:3]),
        comment=f"自动标记: 低分对话 ({evaluation['auto_score']:.0f}分)",
        score_data_accuracy=data_score,
        score_logic=logic_score,
        score_actionability=action_score,
        overall_score=overall_score,
        target_type="conversation",
        target_id=conversation_id,
    )

    logger.info(f"对话 {conversation_id} 自动标记为 Bad Case")
```

### 3.3 高分对话转化为 Eval 基准

**思路**：高分对话可选择性转化为 Eval 用例，作为回归测试基准。

**实现**：

```python
async def _suggest_as_eval_case(conversation_id: int, evaluation: dict):
    """高分对话建议转化为 Eval 用例"""
    from db.conversations import get_messages, get_conversation

    conv = get_conversation(conversation_id)
    messages = get_messages(conversation_id)

    # 提取用户问题
    user_query = ""
    for msg in messages:
        if msg["role"] == "user":
            user_query = msg["content"]
            break

    if not user_query:
        return

    # 生成 Eval 用例建议
    suggestion = {
        "conversation_id": conversation_id,
        "name": f"高质量对话-{conv.get('title', '未命名')[:20]}",
        "analysis_type": "orchestrator",
        "input_params": json.dumps({"question": user_query}, ensure_ascii=False),
        "expected_quality": _generate_expected_quality_from_evaluation(evaluation),
        "auto_score": evaluation.get("auto_score", 0),
    }

    # 保存建议（可在前端展示）
    _save_eval_suggestion(suggestion)

    logger.info(f"对话 {conversation_id} 建议转化为 Eval 用例: {evaluation['auto_score']:.0f}分")


def _generate_expected_quality_from_evaluation(evaluation: dict) -> str:
    """从评估结果生成期望质量标准"""
    breakdown = evaluation.get("auto_score_breakdown", {})
    parts = []

    if breakdown.get("data", 0) >= 80:
        parts.append("引用具体数据和估值指标")
    if breakdown.get("collaboration", 0) >= 80:
        parts.append("多专家协作分析")
    if breakdown.get("response", 0) >= 80:
        parts.append("结构清晰、有风险提示、可操作性强")

    return "；".join(parts) if parts else "专业、准确、可操作的投资分析"
```

### 3.4 评估洞察注入 Prompt 优化

**思路**：将评估发现的系统性问题注入到 Prompt 优化过程中。

**实现**：

```python
async def get_evaluation_insights_for_prompt(agent_key: str = None, days: int = 30) -> str:
    """获取评估洞察，用于 Prompt 优化"""
    from db.eval import get_conversation_eval_stats

    stats = get_conversation_eval_stats()
    if not stats:
        return ""

    insights = []

    # 分析薄弱维度
    avg_scores = stats.get("avg_auto_score", 0)
    if avg_scores < 70:
        insights.append(f"近期对话平均质量偏低: {avg_scores:.0f}分")

    # 分析常见问题
    recent_evals = list_conversation_evaluations(limit=20)
    common_suggestions = _extract_common_suggestions(recent_evals)
    if common_suggestions:
        insights.append("常见改进建议:")
        for suggestion in common_suggestions[:3]:
            insights.append(f"- {suggestion}")

    # 分析专家表现
    expert_issues = _analyze_expert_issues(recent_evals)
    if expert_issues:
        insights.append("专家表现问题:")
        for issue in expert_issues[:3]:
            insights.append(f"- {issue}")

    return "\n".join(insights) if insights else ""


def _extract_common_suggestions(evaluations: list) -> list:
    """提取最常见的改进建议"""
    suggestion_counts = {}
    for eval in evaluations:
        suggestions = eval.get("suggestions", [])
        if isinstance(suggestions, str):
            try:
                suggestions = json.loads(suggestions)
            except:
                suggestions = []

        for s in suggestions:
            suggestion_counts[s] = suggestion_counts.get(s, 0) + 1

    # 按频率排序
    sorted_suggestions = sorted(suggestion_counts.items(), key=lambda x: x[1], reverse=True)
    return [s[0] for s in sorted_suggestions[:5]]
```

### 3.5 专家表现分析与调优

**思路**：分析每个专家在对话中的表现，识别需要调优的专家。

**实现**：

```python
def _analyze_expert_performance(conversation_id: int, evaluation: dict):
    """分析专家表现，识别需要调优的专家"""
    from db.agents import get_agent_runs

    runs = get_agent_runs(conversation_id)
    if not runs:
        return

    expert_stats = {}
    for run in runs:
        agent_key = run.get("agent_key", "")
        if agent_key not in expert_stats:
            expert_stats[agent_key] = {
                "name": run.get("agent_name", ""),
                "total": 0,
                "success": 0,
                "total_duration": 0,
            }

        expert_stats[agent_key]["total"] += 1
        if run.get("status") in ("completed", "success"):
            expert_stats[agent_key]["success"] += 1
        expert_stats[agent_key]["total_duration"] += run.get("duration_ms", 0)

    # 识别表现差的专家
    weak_experts = []
    for agent_key, stats in expert_stats.items():
        success_rate = stats["success"] / stats["total"] if stats["total"] > 0 else 0
        avg_duration = stats["total_duration"] / stats["total"] if stats["total"] > 0 else 0

        if success_rate < 0.8 or avg_duration > 60000:  # 成功率低于80%或平均耗时超过60秒
            weak_experts.append({
                "agent_key": agent_key,
                "name": stats["name"],
                "success_rate": success_rate,
                "avg_duration_ms": avg_duration,
            })

    if weak_experts:
        _save_expert_performance_alert(conversation_id, weak_experts)
```
