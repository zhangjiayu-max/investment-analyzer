## 优化方案

### 1. 查询扩展（Query Expansion）

**目标**：用户输入简短查询时，自动扩展为更完整的查询。

```python
# 当前
query = "PE"
# 扩展后
expanded = "PE 市盈率 估值 百分位 valuation"
```

**实现方式**：

```python
# 同义词词典
SYNONYMS = {
    "PE": ["市盈率", "估值", "valuation", "earnings"],
    "PB": ["市净率", "净资产", "book value"],
    "定投": ["定期定额", "DCA", "dollar cost averaging"],
    "止盈": ["获利了结", "take profit", "profit taking"],
    "回撤": ["下跌", "drawdown", "decline"],
    "夏普": ["夏普比率", "sharpe ratio", "风险调整"],
    "股息": ["分红", "dividend", "派息"],
}

def expand_query(query: str) -> str:
    """扩展查询词。"""
    expanded = [query]
    for word in query.split():
        if word in SYNONYMS:
            expanded.extend(SYNONYMS[word])
    return " ".join(set(expanded))
```

### 2. HyDE (Hypothetical Document Embedding)

**目标**：用 LLM 生成假设性答案，再用答案做向量检索。

```python
# 用户问："沪深300现在能买吗？"
# LLM 生成假设答案：
hypothetical = "沪深300当前PE为11.5，处于历史25%百分位，属于低估区域..."
# 用假设答案做向量检索，找到更相关的文档
```

**实现**：

```python
def hyde_expand(query: str) -> str:
    """生成假设性文档用于检索。"""
    prompt = f"""请用一两句话回答以下投资问题（不需要准确，只需要生成可能的答案）：

问题：{query}

可能的答案："""

    response = _call_llm(
        caller="hyde",
        messages=[{"role": "user", "content": prompt}],
        max_tokens=200,
        temperature=0.7,
    )
    return response.choices[0].message.content
```

### 3. 多路召回融合

**目标**：不同查询使用不同的召回策略。

```python
def multi_route_recall(query: str) -> list[dict]:
    """多路召回融合。"""
    routes = []

    # 路由 1: 原始查询 FTS5
    fts_results = search_fts(query)
    routes.append(("fts_original", fts_results, 1.0))

    # 路由 2: 扩展查询 FTS5
    expanded = expand_query(query)
    if expanded != query:
        fts_expanded = search_fts(expanded)
        routes.append(("fts_expanded", fts_expanded, 0.8))

    # 路由 3: 原始查询向量
    chroma_results = search_chroma(query)
    routes.append(("chroma_original", chroma_results, 1.0))

    # 路由 4: HyDE 向量
    hyde_text = hyde_expand(query)
    chroma_hyde = search_chroma(hyde_text)
    routes.append(("chroma_hyde", chroma_hyde, 0.9))

    # RRF 融合
    return rrf_fusion(routes)
```

### 4. 自适应 RRF 参数

**目标**：根据查询类型调整 RRF 参数。

```python
def adaptive_rrf(query: str, results: list[list]) -> list[dict]:
    """自适应 RRF 融合。"""
    # 分析查询类型
    query_type = classify_query(query)

    # 不同查询类型使用不同的 k 值
    k_map = {
        "factual": 30,      # 事实查询：更信任精确匹配
        "analytical": 60,   # 分析查询：平衡 FTS 和向量
        "conceptual": 100,  # 概念查询：更信任语义相似
    }
    k = k_map.get(query_type, 60)

    return rrf_fusion(results, k=k)
```

```python
# 方案 A: 使用 cross-encoder（精度高，速度慢）
from sentence_transformers import CrossEncoder

reranker = CrossEncoder('cross-encoder/ms-marco-MiniLM-L-6-v2')

def rerank_results(query: str, results: list[dict], top_k: int = 5) -> list[dict]:
    """使用 cross-encoder 重排序。"""
    pairs = [(query, r["body"][:512]) for r in results]
    scores = reranker.predict(pairs)

    for r, score in zip(results, scores):
        r["_rerank_score"] = float(score)

def lightweight_rerank(query: str, results: list[dict], top_k: int = 5) -> list[dict]:
    """轻量级重排序。"""
    query_tokens = set(tokenize(query))

    for r in results:
        body_tokens = set(tokenize(r["body"][:1000]))
        title_tokens = set(tokenize(r["title"]))

        # 计算多种相似度
        body_overlap = len(query_tokens & body_tokens) / max(len(query_tokens), 1)
        title_overlap = len(query_tokens & title_tokens) / max(len(query_tokens), 1)

        # BM25 风格打分
        bm25_score = calculate_bm25(query_tokens, body_tokens)

        # 综合打分
        r["_rerank_score"] = (
            0.4 * body_overlap +
            0.3 * title_overlap +
            0.3 * bm25_score
        )

    return sorted(results, key=lambda x: x["_rerank_score"], reverse=True)[:top_k]
```

### 6. 知识图谱增强

**目标**：利用概念之间的关联关系扩展检索。

```python
# 知识图谱
KNOWLEDGE_GRAPH = {
    "PE": {
        "related": ["PB", "PEG", "股息率", "估值", "百分位"],
        "parent": "估值指标",
        "children": ["PE-TTM", "静态PE", "动态PE"],
    },
    "定投": {
        "related": ["指数基金", "定期定额", "微笑曲线"],
        "parent": "投资策略",
        "children": ["估值定投", "均线定投"],
    },
}

def graph_expand(query: str) -> list[str]:
    """从知识图谱扩展查询。"""
    expanded = []
    for word in tokenize(query):
        if word in KNOWLEDGE_GRAPH:
            node = KNOWLEDGE_GRAPH[word]
            expanded.extend(node.get("related", []))
    return expanded
```
