#### 2.1.2 趋势分计算逻辑

**新增函数**：`_calc_trend_score(fund_code: str) -> tuple[int, str]`

```python
# backend/services/daily_position_advisor.py 新增

def _calc_trend_score(fund_code: str) -> tuple[int, str]:
    """计算近5日净值趋势得分（0-10分）。

    返回 (score, reason)
    - 止跌回升 → 10分
    - 平稳震荡 → 5-7分
    - 持续下跌 → 0-3分
    """
    try:
        from db.market_data import get_fund_nav_history
        navs = get_fund_nav_history(fund_code, days=5)
        if len(navs) < 3:
            return 5, f"近5日数据不足({len(navs)}条)，给中位分"

        # 过滤无效净值
        valid_navs = [n for n in navs if n.get("nav") and n["nav"] > 0]
        if len(valid_navs) < 3:
            return 5, "有效净值不足，给中位分"

        first_nav = valid_navs[0]["nav"]
        last_nav = valid_navs[-1]["nav"]
        recent_ch = (last_nav - first_nav) / first_nav * 100

        # 计算近3日斜率（判断是否止跌）
        if len(valid_navs) >= 4:
            mid_nav = valid_navs[-3]["nav"]
            recent_3d_ch = (last_nav - mid_nav) / mid_nav * 100
        else:
            recent_3d_ch = recent_ch

        # 评分规则
        if recent_ch > 1.0:
            score = 10
            reason = f"近5日 +{recent_ch:.2f}%，止跌回升"
        elif recent_ch > 0:
            score = 7
            reason = f"近5日 +{recent_ch:.2f}%，小幅反弹"
        elif recent_ch > -1.0:
            score = 5
            reason = f"近5日 {recent_ch:.2f}%，横盘震荡"
        elif recent_ch > -2.0:
            score = 3
            reason = f"近5日 {recent_ch:.2f}%，仍在下跌"
        else:
            score = 0
            reason = f"近5日 {recent_ch:.2f}%，持续大跌"

        # 近3日加速下跌 → 再扣2分
        if recent_3d_ch < -1.5:
            score = max(0, score - 2)
            reason += f"（近3日 {recent_3d_ch:.2f}%，加速下跌）"

        return score, reason
    except Exception as e:
        logger.warning(f"趋势分计算异常 {fund_code}: {e}")
        return 5, "趋势计算异常，给中位分"
```

#### 2.1.3 评分函数改造

```python
def _calc_signal_score(
    drop_pct: float,
    val_percentile: float | None,
    position_pct: float,
    cash_pct: float,
    profit_rate: float = 0,
    industry_score: float = 0,
    trend_score: int = 5,        # 新增：近5日趋势分（0-10）
    trend_reason: str = "",      # 新增：趋势说明
) -> tuple[int, str, dict]:
    """信号评分 0-100 + 置信度 + 评分明细。

    维度:
      1. 跌幅 (0-25)          ← 从 35 降为 25
      2. 估值优势 (0-25)
      3. 仓位空间 (0-15)
      4. 流动性 (0-10)
      5. 持仓盈亏修正 (-15 ~ +10)
      6. 行业前景+政策 (0-15)
      7. 近5日趋势 (0-10)     ← 新增
    """
    detail = {}
    score = 0

    # 1. 跌幅（0-25分，原35分）
    s1 = min(drop_pct / 12 * 25, 25) if drop_pct else 0
    detail["跌幅"] = round(s1, 1)
    score += s1

    # 2-6 维度不变...

    # 7. 近5日趋势（0-10分，新增）
    s7 = trend_score
    detail["趋势"] = round(s7, 1)
    score += s7

    final = min(100, max(0, round(score)))

    # 置信度判定（新增趋势分参与）
    if final >= 70 and s5 >= 0 and s6 >= 8 and s7 >= 5:
        confidence = "high"
    elif final >= 55 and s5 >= -10:
        confidence = "medium"
    elif final >= 65:
        confidence = "medium"
    else:
        confidence = "low"

    return final, confidence, detail
```

```python
# backend/services/alert_accuracy_backtest.py

def backtest_alert_accuracy(week_start: str = None):
    """回测过去 7 天所有预警的实际走势。"""
    from datetime import datetime, timedelta
    from db._conn import _get_conn

    if not week_start:
        # 默认回测上周
        today = datetime.now()
        week_start = (today - timedelta(days=today.weekday() + 7)).strftime("%Y-%m-%d")
    week_end = (datetime.strptime(week_start, "%Y-%m-%d") + timedelta(days=7)).strftime("%Y-%m-%d")

    conn = _get_conn()
    try:
        # 查询周内所有涉及基金的预警
        alerts = conn.execute("""
            SELECT alert_type, severity, related_fund_code, created_at
            FROM portfolio_alerts
            WHERE created_at >= ? AND created_at < ?
              AND related_fund_code IS NOT NULL
              AND alert_type IN ('buy_drop_alert', 'drawdown_alert', 'valuation_alert',
                                 'valuation_opportunity', 'concentration_alert')
        """, (week_start, week_end)).fetchall()

        # 按 alert_type + severity 分组统计
        stats = defaultdict(lambda: {"count": 0, "changes": []})
        for a in alerts:
            code = a["related_fund_code"]
            alert_date = a["created_at"][:10]
            # 查预警后 7 天净值变化
            post_navs = conn.execute("""
                SELECT nav FROM fund_nav_history
                WHERE fund_code = ? AND nav_date > ?
                ORDER BY nav_date LIMIT 7
            """, (code, alert_date)).fetchall()
            if len(post_navs) >= 2:
                first = post_navs[0]["nav"]
                last = post_navs[-1]["nav"]
                if first and first > 0:
                    ch = (last - first) / first * 100
                    key = f"{a['alert_type']}:{a['severity']}"
                    stats[key]["count"] += 1
                    stats[key]["changes"].append(ch)

        # 写入统计表
        for key, data in stats.items():
            alert_type, severity = key.split(":")
            changes = data["changes"]
            avg_ch = sum(changes) / len(changes) if changes else 0
            # 胜率：buy_drop/valuation_opportunity 看涨，valuation_alert/concentration 看跌
            if alert_type in ("buy_drop_alert", "valuation_opportunity"):
                wins = sum(1 for c in changes if c > 0)
            else:
                wins = sum(1 for c in changes if c < 0)
            win_rate = wins / len(changes) * 100 if changes else 0

            conn.execute("""
                INSERT INTO alert_accuracy_stats
                (alert_type, severity, week_start, sample_count, avg_followup_change, win_rate, median_change, created_at)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?)
            """, (alert_type, severity, week_start, len(changes), round(avg_ch, 2),
                  round(win_rate, 1), round(sorted(changes)[len(changes)//2] if changes else 0, 2),
                  datetime.now().isoformat()))
        conn.commit()
        logger.info(f"预警准确性回测完成：{len(stats)} 组统计")
    finally:
        conn.close()
```
