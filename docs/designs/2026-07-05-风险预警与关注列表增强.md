```python
def get_alert_history(alert_id: int, days: int = 30) -> list[dict]:
    """查询同持仓同类型预警的历史记录。"""
    conn = _get_conn()
    # 先查当前预警获取 alert_type + related_fund_code
    cur = conn.execute("SELECT * FROM portfolio_alerts WHERE id = ?", (alert_id,)).fetchone()
    if not cur:
        conn.close()
        return []
    cur = dict(cur)
    if not cur.get("related_fund_code"):
        conn.close()
        return []
    rows = conn.execute("""
        SELECT id, severity, title, content, created_at, is_read
        FROM portfolio_alerts
        WHERE alert_type = ? AND related_fund_code = ?
          AND created_at >= datetime('now','localtime', ?)
        ORDER BY created_at DESC LIMIT 20
    """, (cur["alert_type"], cur["related_fund_code"], f"-{days} days")).fetchall()
    conn.close()
    return [dict(r) for r in rows]
```

## 四、P2：买入时机综合评分（轻量版，避免高成本数据源）

### 4.1 买入时机评分卡（P2，轻量版）

**原方案 B 问题**：需要趋势/资金面/北向资金等数据，akshare 调用成本高、数据可能缺失。

**轻量版方案**：只用**已有数据**计算，不新增数据源。

#### 4.1.1 评分维度（4 维，纯本地数据）

| 维度 | 权重 | 数据来源 | 评分逻辑 |
|------|------|---------|---------|
| 估值百分位 | 50% | `current_percentile`（已有） | ≤20%→100分，每升10%扣12分，≥80%→0分 |
| 净值距目标价 | 25% | `current_nav` vs `target_price`（已有） | 达到目标价→100分，距离每增1%扣5分 |
| 与已持仓相关性 | 15% | 持仓相关性矩阵（已有 `analysis/correlation`） | 低相关（<0.3）→100分，高相关（>0.7）→20分 |
| 持仓集中度惩罚 | 10% | 同类基金已持仓占比（已有） | 未持有→100分，已持有且占比>20%→30分 |

**总分 = 估值50% + 净值25% + 相关性15% + 集中度10%**

**评级**：
- ≥75 → 🟢 建议买入
- 50-74 → 🟡 观察
- <50 → 🔴 等待

#### 4.1.2 实现

**新增 API**：`GET /api/watchlist/{item_id}/buy-score`

**文件**：`backend/routers/watchlist.py`

```python
@router.get("/watchlist/{item_id}/buy-score")
async def get_buy_score(item_id: int):
    """计算关注基金的买入时机综合评分（纯规则，无 LLM）。"""
    from db.watchlist import get_watchlist_item
    item = get_watchlist_item(item_id)
    if not item:
        raise HTTPException(404, "关注基金不存在")

    score = _calculate_buy_score(item)
    return {
        "score": score["total"],
        "rating": "buy" if score["total"] >= 75 else "watch" if score["total"] >= 50 else "wait",
        "dimensions": score["dimensions"],
        "calculated_at": datetime.now().isoformat(),
    }
```

**`_calculate_buy_score` 纯函数**（无 LLM、无 MCP、无 akshare）：

```python
def _calculate_buy_score(item: dict) -> dict:
    """计算买入时机评分（纯本地数据，0 外部调用）。"""
    dims = {}

    # 1. 估值百分位（50%）
    pct = item.get("current_percentile")
    if pct is None:
        dims["valuation"] = {"score": 50, "reason": "估值数据缺失"}
    else:
        v_score = max(0, 100 - max(0, pct - 20) * 1.2)
        dims["valuation"] = {"score": round(v_score), "reason": f"PE百分位 {pct:.0f}%"}

    # 2. 净值距目标价（25%）
    nav = item.get("current_nav")
    target = item.get("target_price")
    if not target or not nav or target <= 0:
        dims["price"] = {"score": 50, "reason": "未设目标价"}
    else:
        distance_pct = (nav - target) / target * 100  # 正数=高于目标价
        p_score = max(0, 100 - max(0, distance_pct) * 5)
        dims["price"] = {"score": round(p_score), "reason": f"距目标价 {distance_pct:+.1f}%"}

    # 3. 与已持仓相关性（15%）— 查本地相关性矩阵缓存
    try:
        from routers.analysis.correlation import _get_correlation_cache
        corr = _get_correlation_cache().get(item["fund_code"], {})
        avg_corr = sum(corr.values()) / len(corr) if corr else 0
        c_score = max(20, 100 - avg_corr * 100) if corr else 80
        dims["correlation"] = {"score": round(c_score), "reason": f"平均相关性 {avg_corr:.2f}"}
    except Exception:
        dims["correlation"] = {"score": 80, "reason": "相关性数据缺失"}

    # 4. 持仓集中度惩罚（10%）
    try:
        from db.portfolio import get_holding_by_fund
        h = get_holding_by_fund(item["fund_code"])
        if not h:
            dims["concentration"] = {"score": 100, "reason": "未持有"}
        else:
            # 查该基金在组合中占比
            from db.portfolio import get_portfolio_summary
            summary = get_portfolio_summary()
            ratio = (h.get("current_value") or 0) / max(summary.get("total_value", 1), 1)
            conc_score = max(30, 100 - ratio * 200)  # 占比>35% → 30分
            dims["concentration"] = {"score": round(conc_score), "reason": f"已持仓占比 {ratio:.0%}"}
    except Exception:
        dims["concentration"] = {"score": 80, "reason": "持仓数据缺失"}

    total = (
        dims["valuation"]["score"] * 0.5 +
        dims["price"]["score"] * 0.25 +
        dims["correlation"]["score"] * 0.15 +
        dims["concentration"]["score"] * 0.10
    )
    return {"total": round(total), "dimensions": dims}
```

**成本**：纯本地计算，0 LLM、0 MCP、0 akshare。相关性用缓存，持仓用本地 DB。

```python
    related_fund_code=item["fund_code"],
    related_fund_name=item["fund_name"],
    source="watchlist_patrol",
)
```

**去重**：`create_alert` 已有 24h title+severity 去重。

**成本**：纯数据库，0 LLM。
